:page-role: reference

include::includes/attributes.adoc[]


This cookbook provides a comprehensive collection of practical, copy-and-paste examples for building modern workflows using the Serverless Workflow Java SDK Fluent API.

All workflows in Quarkus Flow are defined as CDI beans by extending the `Flow` class and overriding the `descriptor()` method.

== Essential Setup
Every workflow should follow this structure.

[source,java]
----
package org.acme;

import io.quarkiverse.flow.Flow;
import io.serverlessworkflow.api.types.Workflow;
import io.serverlessworkflow.fluent.func.spec.FuncWorkflowBuilder;
import jakarta.enterprise.context.ApplicationScoped;

// Recommended Static Imports
import static io.serverlessworkflow.fluent.func.dsl.FuncDSL.*;
import static io.serverlessworkflow.fluent.func.spec.FuncWorkflowBuilder.workflow;

@ApplicationScoped
public class MyWorkflow extends Flow {
    @Override
    public Workflow descriptor() {
        return workflow("my-workflow-id")
            .tasks(
                // tasks go here
            )
            .build();
    }
}
----

[#pausing_execution]
== 1. Pausing Execution (Wait State)
To pause a workflow for a specific duration (ISO 8601), use the `wait` task.

[source,java]
----
@ApplicationScoped
public class WaitWorkflow extends Flow {
    @Override
    public Workflow descriptor() {
        return FuncWorkflowBuilder.workflow("wait-duration-8601")
            .tasks(
                waitTask("wait30Seconds").duration("PT30S")
            )
            .build();
    }
}

[#data_transformation]
== 7. Data Transformation
Use the `set` task to initialize or modify the workflow state without external calls

[source,java]
----
@ApplicationScoped
public class SetWorkflow extends Flow {
    @Override
    public Workflow descriptor() {
        return FuncWorkflowBuilder.workflow("set-example")
            .tasks(
                set("initialize",
                    Map.of("processed", true,
                           "timestamp", "${ now() }"
                    )
                )
            )
            .build();
    }
}
----

[#invoking_subflows]
== 11. Invoking Subflows
Keep definitions modular by invoking other workflows using the `run` directive.

[source,java]
----
@ApplicationScoped
public class ForkWorkflow extends Flow {
    @Override
    public Workflow descriptor() {
        return FuncWorkflowBuilder.workflow("fork-example")
            .tasks(
                fork("raiseAlarm")
                    .branch(call("callNurse").http(Method.PUT, "https://hospital.com/nurse"))
                    .branch(call("callDoctor").http(Method.PUT, "https://hospital.com/doctor"))
            )
            .build();
    }
}
----

[#error_handling]
== 12. Error Handling (`try` / `catch`)
Define compensatory logic for business-level errors using `tryTask`.

[source,java]
----
@ApplicationScoped
public class TryCatchWorkflow extends Flow {
    @Override
    public Workflow descriptor() {
        return FuncWorkflowBuilder.workflow("try-catch")
            .tasks(
                tryTask("tryGetPet")
                    .doTasks(
                        call("getPet").http(Method.GET, "https://api.com/pet/{id}")
                    )
                    .catchError(errors -> errors
                        .withType("communication-error")
                        .doTasks(
                            emitJson("notifySupport", "pet.notfound.v1")
                        )
                    )
            )
            .build();
    }
}
----

[#raising_error]
== 13. Raising Explicit Errors
Halt execution and throw specific errors using the `raise` task.

[source,java]
----
@ApplicationScoped
public class GuardWorkflow extends Flow {
    @Override
    public Workflow descriptor() {
        return FuncWorkflowBuilder.workflow("age-gate")
            .tasks(

               switchTask("validateAge")
                   .when(".age < 18", "denyAccess")
                   .defaultTo("proceed"),

               raise("denyAccess")
                   .error(e -> e
                   .type("AccessForbidden")
                   .status(403)
                   .title("Underage Access Attempt")),

               set("proceed", Map.of("authorized", true))
             )
             .build();
    }
}
